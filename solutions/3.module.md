```javascript
//Ex4:
function greetings() {
  var len = arguments.length;
  var names = '';
  for (var i = 0; i < len; i++) {
    names += arguments[i] + ', ';
  }
  return 'Hello, ' + names.slice(0, -2);
}
console.log(greetings('Jhon Doe', 'Jane Doe', 'Jhon Roe', 'Jane Roe'));
```

  var len = arguments.length - 1;

  var names = '';

    names += arguments[i] + ', ';

}
